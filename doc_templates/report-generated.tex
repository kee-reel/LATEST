{
	ARG(1, int, M);
	ARG(2, int, N);
	if(N == 0)
	{
		printf("-");
		return 0;
	}
	char is_negative = M < 0 || N < 0;
	M = M < 0 ? -M : M;
	N = N < 0 ? -N : N;

	int whole = M / N;
	M -= N * whole;
	while(M > 1 && N % M == 0)
	{
		N /= M;
		M /= M;
	}
	if(whole || !M)
	{
		if(is_negative)
			printf("-");
		printf("%d", whole);
	}
	if(M)
	{
		if(whole)
			printf(" ");
		if(is_negative && !whole)
			printf("-");
		printf("%d/%d", M, N);
	}
	return 0;
}
